Erlang R20

mkdir ~/elixir/install ; cd ~/elixir/install
wget -c http://www.erlang.org/download/otp_src_20.3.tar.gz
tar zx < otp_src_20.3.tar.gz
# no out of tree build
cd otp_src_20.3
export ERL_TOP=`pwd`
sudo apt install libncurses5-dev libssl-dev
make distclean
../configure --without-javac --with-ssl --without-wx
make

# 20m53s

sudo make install
